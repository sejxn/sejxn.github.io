### code  
```c++
#include <iostream>
#include <regex>

using namespace std;

int t;

bool Check(string& s)
{
    regex re("(100+1+|01)+");
    return regex_match(s, re);
}

void Solve()
{
    cin >> t;
    while (t--) 
    {
        string s;
        cin >> s;
        if (Check(s))
            cout << "YES" << '\n';
        else
            cout << "NO" << '\n';
    }
}

int main()
{
    cin.tie(NULL);
    ios::sync_with_stdio(false);
    Solve();
    return 0;
}
```
